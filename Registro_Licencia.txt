$dll = "C:\Users\[NombreUsuario]\AppData\Roaming\Autodesk\ApplicationPlugins\EGISParameterSync_Civil3D.bundle\Contents\Win64\EGISParameterSync_Civil3D.dll"

$base = "HKCU:\Software\Autodesk\AutoCAD"
Get-ChildItem $base | ForEach-Object {
    $ver = $_
    Get-ChildItem $ver.PSPath | Where-Object { $_.PSChildName -like "ACAD-*" } | ForEach-Object {
        $prod = $_
        $key = "$($prod.PSPath)\Applications\EGISParameterSync"
        New-Item -Path $key -Force | Out-Null
        Set-ItemProperty -Path $key -Name "DESCRIPTION" -Value "EGIS Parameter Sync Civil 3D"
        Set-ItemProperty -Path $key -Name "LOADCTRLS"   -Value 14 -Type DWord
        Set-ItemProperty -Path $key -Name "LOADER"      -Value $dll
        Set-ItemProperty -Path $key -Name "MANAGED"     -Value 1  -Type DWord
        Write-Host "OK: $($ver.PSChildName) / $($prod.PSChildName)" -ForegroundColor Green
    }
}
Write-Host "Listo. Reinicia Civil 3D." -ForegroundColor Yellow
